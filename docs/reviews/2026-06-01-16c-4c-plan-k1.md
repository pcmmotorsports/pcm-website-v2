   ```ts
   // NEXT_PUBLIC_SITE_URL 設絕對 http(s) → 用(去尾斜線);未設/格式不合且 dev → localhost;production → undefined(省略 URL 欄,絕不在 prod 吐 localhost / 壞值 canonical)
   export function resolveSiteUrl(): string | undefined {
     const env = process.env.NEXT_PUBLIC_SITE_URL?.trim();
     if (env && isAbsoluteHttpUrl(env)) return env.replace(/\/+$/, '');
     return process.env.NODE_ENV === 'production' ? undefined : 'http://localhost:3000';
   }
   ```

## 4. JSON-LD 結構(general only)

```
{
  '@context':'https://schema.org', '@type':'Product',
  name,                                   // product.name
  image,                                  // product.images 過 isAbsoluteHttpUrl guard 才放(相對 placeholder 濾掉)
  description,                            // product.subtitle?.trim() || product.name
  sku,                                    // product.productCode(有才放)
  brand: { '@type':'Brand', name: product.brand },   // RPM-only 期 = "RPM CARBON"(資料驅動、非 hardcode)
  category,                               // product.category(raw 字串)
  offers:
    多變體且價有高低 → { '@type':'AggregateOffer', priceCurrency:'TWD', lowPrice, highPrice, offerCount }
    其餘(單一/變體同價)→ { '@type':'Offer', priceCurrency:'TWD', price }
    // 最終:不放 availability(Q3=A、#161 不顯庫存);offers 僅 general 價、無 itemCondition
  url                                     // canonical(resolveSiteUrl 有值才放)
}
```
